% 4. Fundamentals: Constructing the Network
%---------------------------------------------------------
\begin{frame}{Fundamentals: Constructing the Network}
    \textbf{The Dataset:} 1,908 ChatGPT conversations (Dec '22 -- Apr '25).
    
    \vspace{0.5cm}
    
    \textbf{Two Methodological Challenges:}
    
    \begin{enumerate}
        \item \textbf{Representation:} We used \texttt{nomic-embed-text} (8k context) to vectorise whole conversations.
        
        \vspace{0.3cm}
        
        \item \textbf{The "Chatty" Problem:} AI responses are verbose.
        \begin{itemize}
            \item \textit{Solution:} \textbf{User-Weighted Embeddings}.
            \item We weight the User's prompt $\mathbf{2:1}$ over the AI's response.
            \item This prioritizes \emph{intent} over \emph{content generation}.
        \end{itemize}
    \end{enumerate}

    \vspace{0.2cm}
    \footnotesize{\textit{*Ablation studies confirmed 2:1 at $\theta=0.9$ similarity maximizes community modularity.}}
\end{frame}

% 6. Insight 1: Heterogeneous Topology
%---------------------------------------------------------
\begin{frame}{Insight 1: Not All Knowledge "Looks" the Same}
    We found two distinct structural archetypes for knowledge exploration:
    
    \vspace{0.5cm}
    
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{Theoretical Domains}
            \textit{(e.g., ML Theory, Math, Philosophy)}
            \begin{itemize}
                \item \textbf{Hub-and-Spoke}
                \item High clustering coefficient.
                \item \textbf{Meaning:} Concepts anchor to central theories.
            \end{itemize}
        \end{block}
        
        \column{0.5\textwidth}
        \begin{block}{Practical Domains}
            \textit{(e.g., Programming Projects)}
            \begin{itemize}
                \item \textbf{Tree-like / Branching}
                \item Low clustering coefficient.
                \item \textbf{Meaning:} Task-specific problem solving; linear exploration without central hubs.
            \end{itemize}
        \end{block}
    \end{columns}
\end{frame}

%---------------------------------------------------------
% 7. Insight 2: The Bridge Taxonomy
%---------------------------------------------------------
\begin{frame}{Insight 2: A Taxonomy of "Bridges"}
    How does the mind jump between domains? We identified three bridge types via centrality analysis:
    
    \vspace{0.3cm}
    
    \begin{itemize}
        \item \textbf{1. Evolutionary Bridges (Topic Drift)}
        \begin{itemize}
            \item \small{High Degree, High Betweenness.}
            \item \textit{Example:} Starting with "Geometric Mean" $\rightarrow$ Drifting to "Loss Functions".
        \end{itemize}
        
        \item \textbf{2. Integrative Bridges (Synthesis)}
        \begin{itemize}
            \item \small{Moderate Degree.}
            \item \textit{Example:} "MCTS Code" connecting \textit{AI Theory} with \textit{Software Engineering}.
        \end{itemize}
        
        \item \textbf{3. Pure Bridges (Weak Ties)}
        \begin{itemize}
            \item \small{Low Degree, High Importance.}
            \item \textit{Example:} A single conversation on "CUDA Compilation" linking \textit{SysAdmin} to \textit{Dev}.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Conclusion & The Future}
    \textbf{Summary:}
    \begin{itemize}
        \item AI logs are a rich, untapped source of \textbf{cognitive data}.
        \item We can map them using \textbf{User-Weighted Embeddings}.
        \item Knowledge topology is \textbf{heterogeneous} (Trees vs. Hubs).
    \end{itemize}
    
    \vspace{0.5cm}
    
    \textbf{The Promise:}
    \begin{itemize}
        \item \textbf{For Network Science:} A new dataset for studying "Exploratory" networks vs "Citation" networks.
        \item \textbf{For HCI:} Better retrieval systems that respect \textit{semantic boundaries} and \textit{bridges}.
    \end{itemize}
